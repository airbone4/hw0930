import ollama

# 設定模型名稱（請確保已執行過 ollama pull llama3）
model_name =  'llama3.2:1b'  

def chat_simple():
    
    
    #question = '為什麼天空是藍色的？用一句話回答。'
    question = input('輸入問題:')
    if question == "exit":
        quit()
    response = ollama.chat(model=model_name, messages=[
        {
            'role': 'user',
            'content': question,
        },
    ])
    print(response['message']['content'])
 

if __name__ == "__main__":
  while True:  
    chat_simple()
