var 宣告 let 宣告 哪裡不一樣

let 的生命範圍是block ,但是var 的生命範圍 不是block 而是函數block

看下面的程式碼註解

        // var y=5;  //註解1:
        {// 這是block 但不是函數block
        let x=3;
        var y=5;  //在這裡宣告,等於在註解1宣告。因為var 宣告的y ,其生命期只受函數block 影響
        }

        console.log(y)
        console.log(x)

        //函數sayHello的定義
        function sayHello(msg) 
        { //這是函數block
            let x=4;  //函數結束,x消失 (結束生命期)
            var y=5; //和x一樣 也是,函數消失就結束。因為 var 宣告的變數,受函數block影響。
            let output= document.getElementById('output');
            output.innerHTML = msg+'x: '+x+'y: '+y;
        }        

        sayHello('hello ') //調用函數sayHello
